void mouseClicked(){
  if (value == 0 && mouseX > 335 && mouseX < 465 && mouseY > 335 && mouseY < 465) {
    value = 300;
  } else if(value == 300 && mouseX > 335 && mouseX < 465 && mouseY > 335 && mouseY < 465) {
    value = 0;
  }
  
  if (value1 == 0 && mouseX > 500 && mouseX < 800 && mouseY > 400 && mouseY < 600) {
    value1 = 300;
  } else if(value1 == 300 && mouseX > 600 && mouseX < 800 && mouseY > 400 && mouseY < 600) {
    value1 = 0;
  }
  
  if (value2 == #02AF97 && mouseX > 200 && mouseX < 300 && mouseY > 500 && mouseY < 600) {
    value2 = 0;
  } else if(value2 == 0 && mouseX > 200 && mouseX < 300 && mouseY > 500 && mouseY < 600) {
    value2 = #02AF97;
  }
  
  if (value3 == 0 && value4 == 0 && value5 == 0 && mouseX > 550 && mouseX < 780 && mouseY > 120 && mouseY < 380){
    value3 = 7;
    value4 = 255;
    value5 = 1;
  } else if(value3 == 7 && value4 == 255 && value5 == 1 && mouseX > 550 && mouseX < 780 && mouseY > 120 && mouseY < 380){
    value3 = 0;
    value4 = 0;
    value5 = 0;
  }
  
  if (value6 == 0 && value7 ==0 && mouseX > 100 && mouseX < 300 && mouseY > 250 && mouseY < 400) {
    value6 = 1;
    value7 = 300;
  } else if(value6 == 1 && value7 == 300 && mouseX > 100 && mouseX < 300 && mouseY > 250 && mouseY < 400) {
    value6 = 0;
    value7 = 0;
  }

void mouseDragged() {
  float rate = 0.01;
  rotx += (pmouseY-mouseY) * rate;
  roty += (mouseX-pmouseX) * rate;
}

class Star {
  float aa;
  float bb;
  float cc;
  
  float pcc;
  
  Star(){
    aa = random(-width, width);
    bb = random(-height, height);
    cc = random(width);
    
    pcc = cc;
  }
  void update(){
    cc = cc - 20;
    if(cc <1){
      cc = width;
      aa = random(-width, width);
      bb = random(-height, height);
      pcc = cc;
    }
  }
  
  void show(){
    fill(255);
    noStroke();
    
    float saa = map(aa/cc, 0, 1, 0, width);
    float sbb = map(bb/cc, 0, 1, 0, height);
    
    float r = map(cc, 0, width, 16, 0);
    //ellipse(saa, sbb, r, r);
    
    float paa = map(aa/pcc, 0, 1, 0, width);
    float pbb = map(bb/pcc, 0, 1, 0, height);
    
    pcc = cc;
    
    pushMatrix();
    stroke(value8);
    strokeWeight(value11);
    line(paa, pbb, saa, sbb);
    popMatrix();
  }
}
